clearsysmsg

if not varexist 'mytarget'
  overhead 'Select your target'
  setvar! 'mytarget'
endif

createtimer 'RefreshFireEle'

while not dead
  cast 'Fire Elemental'
  wait 5000
  say 'all kill'
  wft 250
  target 'mytarget'

  if timer 'RefreshFireEle' > 4000
    overhead 'Attempting to focus aggression"
    dclicktype 3713
    wft 250
    target 'mytarget'
    settimer 'RefreshFireEle' 0
  endif
loop  
endwhile

if dead
  unsetvar 'mytarget'
  stop
endif
